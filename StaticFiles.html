Static Files

Serving Static Files

We can use the Express to serve static files as well.

First we need to create some files for the server.

index.html

<!doctype html>
<html>
  <!-- FILL IN THINGS HERE -->
</html>

Respond with static file

static_file.js

const express = require("express"); // import express framework.

const app = express();

// serve response from file
app.get("/", (req, res) => {
  res.sendFile("./static/index.html");
});

// start the server on port 8080
app.listen(8080, () => {
  console.log("Server listening on port 8080");
});

Respond with files in a directory

static_dir.js

const express = require("express"); // import express framework.

const app = express();

// serve all files from the `static' directory
app.use(express.static("static"));

// start the server on port 8080
app.listen(8080, () => {
  console.log("Server listening on port 8080");
});
